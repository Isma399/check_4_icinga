cmake_minimum_required(VERSION 2.6)
project(Check_linux_disk)

if (CMAKE_BINARY_DIR STREQUAL ${CMAKE_SOURCE_DIR})
    message(FATAL_ERROR "Building in source dir prohibited. Remove CMakeCache.txt and try 'mkdir build && cd build && cmake ..'.")
endif ()
if (NOT CMAKE_BUILD_TYPE OR CMAKE_BUILD_TYPE EQUAL "")
    message(STATUS "No build type selected, default to RelWithDebInfo")
    set(CMAKE_BUILD_TYPE "RelWithDebInfo")
endif ()

include(FindNetSnmp.cmake)
include_directories(${CNETSNMP_INCLUDE_DIR})

add_executable(Check_linux_disk  check_linux_disk.c)
target_link_libraries(Check_linux_disk ${NETSNMP_LIBRARIES})
install(TARGETS ${TARGETS} DESTINATION bin)
